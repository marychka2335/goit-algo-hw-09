# –§—É–Ω–∫—Ü—ñ—è –∂–∞–¥—ñ–±–Ω–æ–≥–æ –∞–ª–≥–æ—Ä–∏—Ç–º—É –¥–ª—è –∫–∞—Å–æ–≤–æ—ó —Å–∏—Å—Ç–µ–º–∏, —è–∫–∞ –≤–∏–¥–∞—î —Ä–µ—à—Ç—É –ø–æ–∫—É–ø—Ü–µ–≤—ñ
def find_coins_greedy(amount):
    result = {}
    for coin in coins:
        if amount >= coin:
            result[coin] = amount // coin
            amount = amount % coin
    return result

print("–ñ–∞–¥—ñ–±–Ω–∏–π –∞–ª–≥–æ—Ä–∏—Ç–º:", find_coins_greedy(113))  # {50: 2, 10: 1, 2: 1, 1: 1}

# –§—É–Ω–∫—Ü—ñ—è –¥–∏–Ω–∞–º—ñ—á–Ω–æ–≥–æ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—è –¥–ª—è –∫–∞—Å–æ–≤–æ—ó —Å–∏—Å—Ç–µ–º–∏, —è–∫–∞ –≤–∏–¥–∞—î —Ä–µ—à—Ç—É –ø–æ–∫—É–ø—Ü–µ–≤—ñ
def find_min_coins(amount):
    # –Ü–Ω—ñ—Ü—ñ–∞–ª—ñ–∑–∞—Ü—ñ—è —Ç–∞–±–ª–∏—Ü—ñ –¥–ª—è –∑–±–µ—Ä—ñ–≥–∞–Ω–Ω—è –º—ñ–Ω—ñ–º–∞–ª—å–Ω–æ—ó –∫—ñ–ª—å–∫–æ—Å—Ç—ñ –º–æ–Ω–µ—Ç –¥–ª—è –∫–æ–∂–Ω–æ—ó —Å—É–º–∏
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0  # –î–ª—è —Å—É–º–∏ 0 –ø–æ—Ç—Ä—ñ–±–Ω–æ 0 –º–æ–Ω–µ—Ç
    # –¢–∞–±–ª–∏—Ü—è –¥–ª—è –∑–±–µ—Ä—ñ–≥–∞–Ω–Ω—è –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–∏—Ö –º–æ–Ω–µ—Ç
    coin_used = [-1] * (amount + 1)
    # –ó–∞–ø–æ–≤–Ω–µ–Ω–Ω—è —Ç–∞–±–ª–∏—Ü—ñ dp
    for i in range(1, amount + 1):
        for coin in coins:
            if i >= coin and dp[i - coin] + 1 < dp[i]:
                dp[i] = dp[i - coin] + 1
                coin_used[i] = coin
    # –Ø–∫—â–æ —Å—É–º–∞ –Ω–µ –º–æ–∂–µ –±—É—Ç–∏ —Å—Ñ–æ—Ä–º–æ–≤–∞–Ω–∞ –∑ –¥–æ—Å—Ç—É–ø–Ω–∏—Ö –º–æ–Ω–µ—Ç
    if dp[amount] == float('inf'):
        return {}
    # –í—ñ–¥–Ω–æ–≤–ª–µ–Ω–Ω—è –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–∏—Ö –º–æ–Ω–µ—Ç
    result = {}
    i = amount
    while i > 0:
        coin = coin_used[i]
        if coin in result:
            result[coin] += 1
        else:
            result[coin] = 1
        i -= coin
    return result

print("–î–∏–Ω–∞–º—ñ—á–Ω–µ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—è:", find_min_coins(113))  # {50: 2, 10: 1, 2: 1, 1: 1}

# –ü–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è —á–∞—Å—É –≤–∏–∫–æ–Ω–∞–Ω–Ω—è –∂–∞–¥—ñ–±–Ω–æ–≥–æ –∞–ª–≥–æ—Ä–∏—Ç–º—É —Ç–∞ –¥–∏–Ω–∞–º—ñ—á–Ω–æ–≥–æ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—è
amounts = [15, 60, 113, 129, 130, 200, 500, 999, 1005]
results = []

for amount in amounts:
    time_greedy = timeit.timeit(lambda: find_coins_greedy(amount), number=1000)
    time_dp = timeit.timeit(lambda: find_min_coins(amount), number=1000)
    results.append([amount, time_greedy, time_dp])

print("| Amount | Greedy Time (s) | DP Time (s) |")
print("|--------|-----------------|-------------|")
for result in results:
    print(f"| {result[0]:>6} | {result[1]:>15.8f} | {result[2]:>11.8f} |")
